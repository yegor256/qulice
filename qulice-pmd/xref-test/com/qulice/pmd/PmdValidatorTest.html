112     /**
113      * PmdValidator can find violations in Java file(s).
114      * @throws Exception If something wrong happens inside.
115      */
116     @Test
117     void findsProblemsInJavaFiles() throws Exception {
118         final String file = "src/main/java/Main.java";
119         final Environment env = new Environment.Mock()
120             .withFile(file, "class Main { int x = 0; }");
121         MatcherAssert.assertThat(
122             new PmdValidator(env).validate(
123                 Collections.singletonList(new File(env.basedir(), file))
124             ),
125             Matchers.not(Matchers.<Violation>empty())
126         );
127     }

145     /**
146      * PmdValidator does not think that constant is unused when it is used
147      * just from the inner class.
148      * @throws Exception If something wrong happens inside.
149      */
150     @Test
151     @SuppressWarnings("PMD.AvoidDuplicateLiterals")
152     void doesNotComplainAboutConstantsInInnerClasses() throws Exception {
153         final String file = "src/main/java/foo/Foo.java";
154         final Environment env = new Environment.Mock().withFile(
155             file,
156             Joiner.on('\n').join(
157                 "package foo;",
158                 "interface Foo {",
159                 "  final class Bar implements Foo {",
160                 "    private static final Pattern TEST =",
161                 "      Pattern.compile(\"hey\");",
162                 "    public String doSomething() {",
163                 "      return Foo.Bar.TEST.toString();",
164                 "    }",
165                 "  }",
166                 "}"
167             )
168         );
169         MatcherAssert.assertThat(
170             new PmdValidator(env).validate(
171                 Collections.singletonList(new File(env.basedir(), file))
172             ),
173             Matchers.<Violation>empty()
174         );
175     }
