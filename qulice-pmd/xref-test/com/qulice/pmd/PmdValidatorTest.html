86      /**
87       * PmdValidator can find violations in Java file(s).
88       * @throws Exception If something wrong happens inside.
89       */
90      @Test
91      void findsProblemsInJavaFiles() throws Exception {
92          final String file = "src/main/java/Main.java";
93          final Environment env = new Environment.Mock()
94              .withFile(file, "class Main { int x = 0; }");
95          MatcherAssert.assertThat(
96              "Violations should be found",
97              new PmdValidator(env).validate(
98                  Collections.singletonList(new File(env.basedir(), file))
99              ),
100             Matchers.not(Matchers.<Violation>empty())
101         );
102     }

120     /**
121      * PmdValidator does not think that constant is unused when it is used
122      * just from the inner class.
123      * @throws Exception If something wrong happens inside.
124      */
125     @Test
126     @SuppressWarnings("PMD.AvoidDuplicateLiterals")
127     void doesNotComplainAboutConstantsInInnerClasses() throws Exception {
128         final String file = "src/main/java/foo/Foo.java";
129         final Environment env = new Environment.Mock().withFile(
130             file,
131             Joiner.on('\n').join(
132                 "package foo;",
133                 "interface Foo {",
134                 "  final class Bar implements Foo {",
135                 "    private static final Pattern TEST =",
136                 "      Pattern.compile(\"hey\");",
137                 "    public String doSomething() {",
138                 "      return Foo.Bar.TEST.toString();",
139                 "    }",
140                 "  }",
141                 "}"
142             )
143         );
144         MatcherAssert.assertThat(
145             "Private constant in inner class is not a violation",
146             new PmdValidator(env).validate(
147                 Collections.singletonList(new File(env.basedir(), file))
148             ),
149             Matchers.<Violation>empty()
150         );
151     }
