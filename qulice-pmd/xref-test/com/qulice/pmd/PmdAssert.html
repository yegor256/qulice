5   package com.qulice.pmd;
6   
7   import com.qulice.spi.Environment;
8   import com.qulice.spi.Violation;
9   import java.io.File;
10  import java.util.Collection;
11  import java.util.Collections;
12  import org.cactoos.text.TextOf;
13  import org.hamcrest.Matcher;
14  import org.hamcrest.MatcherAssert;
15  
16  /**
17   * PMD Validator assertions.
18   * @since 0.16
19   */
20  final class PmdAssert {
21      /**
22       * File to validate.
23       */
24      private final String file;
25  
26      /**
27       * Expected build status, true means success.
28       */
29      private final Matcher<Boolean> result;
30  
31      /**
32       * Matcher that needs to match.
33       */
34      private final Matcher<String> matcher;
35  
36      /**
37       * Constructor.
38       * @param file File to validate.
39       * @param result Expected build status.
40       * @param matcher Matcher that needs to match.
41       */
42      PmdAssert(final String file, final Matcher<Boolean> result,
43          final Matcher<String> matcher) {
44          this.file = file;
45          this.result = result;
46          this.matcher = matcher;
47      }
48  
49      /**
50       * Validates given file against PMD.
51       * @throws Exception In case of error.
52       */
53      public void validate() throws Exception {
54          final Environment.Mock mock = new Environment.Mock();
55          final String name = String.format("src/main/java/foo/%s", this.file);
56          final Environment env = mock.withFile(
57              name,
58              new TextOf(
59                  this.getClass().getResourceAsStream(this.file)
60              ).asString()
61          );
62          final Collection<Violation> violations = new PmdValidator(env).validate(
63              Collections.singletonList(new File(env.basedir(), name))
64          );
65          MatcherAssert.assertThat(violations.isEmpty(), this.result);
66          final StringBuilder builder = new StringBuilder();
67          for (final Violation violation : violations) {
68              builder.append(
69                  String.format(
70                      "PMD: %s[%s]: %s (%s)\n",
71                      this.file,
72                      violation.lines(),
73                      violation.message(),
74                      violation.name()
75                  )
76              );
77          }
78          MatcherAssert.assertThat(builder.toString(), this.matcher);
79      }
80  }
