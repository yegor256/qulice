118     /**
119      * PmdValidator does not think that constant is unused when it is used just
120      * from the inner class.
121      *
122      * @throws Exception If something wrong happens inside.
123      */
124     @Test
125     @SuppressWarnings("PMD.AvoidDuplicateLiterals")
126     void doesNotComplainAboutConstantsInInnerClasses() throws Exception {
127         final String file = "src/main/java/foo/Foo.java";
128         final Environment env = new Environment.Mock().withFile(
129             file,
130             Joiner.on('\n').join(
131                 "package foo;",
132                 "interface Foo {",
133                 "  final class Bar implements Foo {",
134                 "    private static final Pattern TEST =",
135                 "      Pattern.compile(\"hey\");",
136                 "    public String doSomething() {",
137                 "      return Foo.Bar.TEST.toString();",
138                 "    }",
139                 "  }",
140                 "}"
141             )
142         );
143         MatcherAssert.assertThat(
144             "Private constant in inner class is not a violation",
145             new PmdValidator(env).validate(
146                 Collections.singletonList(new File(env.basedir(), file))
147             ),
148             Matchers.<Violation>empty()
149         );
150     }
