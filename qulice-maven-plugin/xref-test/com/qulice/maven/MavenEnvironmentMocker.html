27  /**
28   * Mocker of {@link MavenProject}.
29   * @since 0.4
30   */
31  @SuppressWarnings({
32      "PMD.TooManyMethods",
33      "PMD.ConstructorOnlyInitializesOrCallOtherConstructors"
34  })
35  public final class MavenEnvironmentMocker {
36  
37      /**
38       * Env mocker.
39       */
40      private final Environment.Mock ienv;
41  
42      /**
43       * Project.
44       */
45      private MavenProjectMocker prj;
46  
47      /**
48       * Plexus container, mock.
49       */
50      private final PlexusContainer container;
51  
52      /**
53       * Xpath queries to test pom.xml.
54       */
55      private Collection<String> ass;
56  
57      /**
58       * Public ctor.
59       * @throws IOException If some IO problem inside
60       */
61      public MavenEnvironmentMocker() throws IOException {
62          this.prj = new MavenProjectMocker();
63          this.ass = new LinkedList<>();
64          this.ienv = new Environment.Mock();
65          try {
66              this.container = new DefaultPlexusContainer();
67          } catch (final PlexusContainerException ex) {
68              throw new IOException(ex);
69          }
70      }
71  
72      /**
73       * Inject this object into plexus container.
74       * @param role The role
75       * @param hint The hint
76       * @param object The object to return
77       * @return This object
78       * @throws Exception If something wrong happens inside
79       */
80      public MavenEnvironmentMocker inPlexus(final String role, final String hint,
81          final Object object) throws Exception {
82          this.container.addComponent(object, role);
83          return this;
84      }
85  
86      /**
87       * With this project mocker.
88       * @param mocker The project mocker
89       * @return This object
90       */
91      public MavenEnvironmentMocker with(final MavenProjectMocker mocker) {
92          this.prj = mocker;
93          return this;
94      }
95  
96      /**
97       * With this file on board.
98       * @param name File name related to basedir
99       * @param content File content to write
100      * @return This object
101      * @throws IOException If some IO problem
102      */
103     public MavenEnvironmentMocker withFile(final String name,
104         final String content) throws IOException {
105         this.ienv.withFile(name, content);
106         return this;
107     }
108 
109     /**
110      * With this file on board.
111      * @param name File name related to basedir
112      * @param bytes File content to write
113      * @return This object
114      * @throws IOException If some IO problem
115      */
116     public MavenEnvironmentMocker withFile(final String name,
117         final byte[] bytes) throws IOException {
118         this.ienv.withFile(name, bytes);
119         return this;
120     }
121 
122     /**
123      * With list of xpath queries to validate pom.xml.
124      * @param asserts Collection of xpath queries
125      * @return This object
126      */
127     public MavenEnvironmentMocker withAsserts(
128         final Collection<String> asserts) {
129         this.ass = Collections.unmodifiableCollection(asserts);
130         return this;
131     }
132 
133     /**
134      * Mock it.
135      * @return The environment just mocked
136      * @throws Exception If something wrong happens inside
137      */
138     public MavenEnvironment mock() throws Exception {
139         StaticLoggerBinder.getSingleton().setMavenLog(
140             new DefaultLog(
141                 new ConsoleLogger()
142             )
143         );
144         this.prj.inBasedir(this.ienv.basedir());
145         final MavenProject project = this.prj.mock();
146         final MavenEnvironment env = new FakeMavenEnvironment(
147             project,
148             new FakeContext(this.container),
149             this.ass
150         );
151         return new MavenEnvironment.Wrap(this.ienv, env);
152     }

154     /**
155      * FakeContext.
156      * A mock to a context.

158      * @since 0.24.1
159      */
160     private static final class FakeContext implements Context {
161         /**
162          * Container.
163          */
164         private final PlexusContainer container;
165 
166         FakeContext(final PlexusContainer ctainer) {
167             this.container = ctainer;
168         }
169 
170         @Override
171         public boolean contains(final Object obj) {
172             return true;
173         }
174 
175         @Override
176         public void put(final Object obja, final Object objb) {
177             // Intentionally left blank
178         }
179 
180         @Override
181         public Object get(final Object obj) throws ContextException {
182             return this.container;
183         }
184 
185         @Override
186         public Map<Object, Object> getContextData() {
187             return Collections.emptyMap();
188         }
189     }
