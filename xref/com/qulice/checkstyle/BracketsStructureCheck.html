30  package com.qulice.checkstyle;
31  
32  import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
33  import com.puppycrawl.tools.checkstyle.api.DetailAST;
34  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
35  
36  /**
37   * Checks node/closing brackets to be the last symbols on the line.
38   *
39   * <p>This is how a correct bracket structure should look like:
40   *
41   * <pre>
42   * String text = String.format(
43   *   "some text: %s",
44   *   new Foo().with(
45   *     "abc",
46   *     "foo"
47   *   )
48   * );
49   * </pre>
50   *
51   * <p>The motivation for such formatting is simple - we want to see the entire
52   * block as fast as possible. When you look at a block of code you should be
53   * able to see where it starts and where it ends. In exactly the same way
54   * we organize curled brackets.
55   *
56   * <p>In other words, when you open a bracket and can't close it at the same
57   * line - you should leave it as the last symbol at this line.
58   *
59   * @since 0.3
60   */
61  public final class BracketsStructureCheck extends AbstractCheck {
62  
63      @Override
64      public int[] getDefaultTokens() {
65          return new int[] {
66              TokenTypes.LITERAL_NEW,
67              TokenTypes.METHOD_CALL,
68          };
69      }
70  
71      @Override
72      public void visitToken(final DetailAST ast) {
73          if (ast.getType() == TokenTypes.METHOD_CALL
74              || ast.getType() == TokenTypes.LITERAL_NEW) {
75              this.checkParams(ast);
76          } else {
77              final DetailAST brackets = ast.findFirstToken(TokenTypes.LPAREN);
78              if (brackets != null) {
79                  this.checkParams(brackets);
80              }
81          }
82      }
83  
84      /**
85       * Checks params statement to satisfy the rule.
86       * @param node Tree node, containing method call statement.
87       */
88      private void checkParams(final DetailAST node) {
89          final DetailAST closing = node.findFirstToken(TokenTypes.RPAREN);
90          if (closing != null) {
91              this.checkLines(node, node.getLineNo(), closing.getLineNo());
92          }
93      }
94  
95      /**
96       * Checks params statement to satisfy the rule.
97       * @param node Tree node, containing method call statement.
98       * @param start First line
99       * @param end Final line
100      */
101     private void checkLines(final DetailAST node, final int start,
102         final int end) {
103         if (start != end) {
104             final DetailAST elist = node.findFirstToken(TokenTypes.ELIST);
105             final int pline = elist.getLineNo();
106             if (pline == start) {
107                 this.log(pline, "Parameters should start on a new line");
108             }
109             this.checkExpressionList(elist, end);
110         }
111     }
112 
113     /**
114      * Checks expression list if closing bracket is on new line.
115      * @param elist Tree node, containing expression list
116      * @param end Final line
117      */
118     private void checkExpressionList(final DetailAST elist, final int end) {
119         if (elist.getChildCount() > 0) {
120             final DetailAST last = elist.getLastChild();
121             final int lline = last.getLineNo();
122             if (lline == end) {
123                 this.log(lline, "Closing bracket should be on a new line");
124             }
125         }
126     }
127 
128 }
