107     /**
108      * Test checkstyle for true positive.
109      * @throws Exception If something goes wrong
110      */
111     @Test
112     public void testCheckstyleTruePositive() throws Exception {
113         final AuditListener listener = Mockito.mock(AuditListener.class);
114         final Collector collector = new ChecksTest.Collector();
115         Mockito.doAnswer(collector).when(listener)
116             .addError(Mockito.any(AuditEvent.class));
117         this.check("/Invalid.java", listener);
118         final String[] violations = StringUtils.split(
119             IOUtils.toString(
120                 this.getClass().getResourceAsStream(
121                     String.format("%s/violations.txt", this.dir)
122                 )
123             ),
124             "\n"
125         );
126         for (String line : violations) {
127             final String[] sectors = StringUtils.split(line, ":");
128             final Integer pos = Integer.valueOf(sectors[0]);
129             final String needle = sectors[1].trim();
130             MatcherAssert.assertThat(
131                 collector.has(pos, needle),
132                 Matchers.describedAs(
133                     String.format(
134                         "Line no.%d ('%s') not reported by %s: '%s'",
135                         pos,
136                         needle,
137                         this.dir,
138                         collector.summary()
139                     ),
140                     Matchers.is(true)
141                 )
142             );
143         }
144     }

162     /**
163      * Mocked collector of checkstyle events.
164      */
165     private static class Collector implements Answer<Object> {
166         /**
167          * List of events received.
168          */
169         private final transient List<AuditEvent> events =
170             new ArrayList<AuditEvent>();
171         /**
172          * {@inheritDoc}
173          */
174         @Override
175         public Object answer(final InvocationOnMock invocation) {
176             this.events.add((AuditEvent) invocation.getArguments()[0]);
177             return null;
178         }
179         /**
180          * Do we have this message for this line?
181          * @param line The number of the line
182          * @param msg The message we're looking for
183          * @return This message was reported for the give line?
184          */
185         public boolean has(final Integer line, final String msg) {
186             boolean has = false;
187             for (AuditEvent event : this.events) {
188                 if (event.getLine() == line && event.getMessage().equals(msg)) {
189                     has = true;
190                     break;
191                 }
192             }
193             return has;
194         }
195         /**
196          * Returns full summary.
197          * @return The test summary of all events
198          */
199         public String summary() {
200             final List<String> msgs = new ArrayList<String>();
201             for (AuditEvent event : this.events) {
202                 msgs.add(
203                     String.format(
204                         "%s:%s",
205                         event.getLine(),
206                         event.getMessage()
207                     )
208                 );
209             }
210             return StringUtils.join(msgs, "; ");
211         }
212     }

214     /**
215      * Check one file.
216      * @param name The name of the check
217      * @param listener The listener
218      * @throws Exception If something goes wrong inside
219      */
220     private void check(final String name, final AuditListener listener)
221         throws Exception {
222         final Checker checker = new Checker();
223         final InputSource src = new InputSource(
224             this.getClass().getResourceAsStream(
225                 String.format("%s/config.xml", this.dir)
226             )
227         );
228         checker.setClassloader(Thread.currentThread().getContextClassLoader());
229         checker.setModuleClassLoader(
230             Thread.currentThread().getContextClassLoader()
231         );
232         checker.configure(
233             ConfigurationLoader.loadConfiguration(
234                 src,
235                 new PropertiesExpander(new Properties()),
236                 true
237             )
238         );
239         final List<File> files = new ArrayList<File>();
240         files.add(
241             new File(this.getClass().getResource(this.dir + name).getFile())
242         );
243         checker.addListener(listener);
244         checker.process(files);
245         checker.destroy();
246     }
