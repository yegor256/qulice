783     /**
784      * Returns string with Checkstyle validation results.
785      * @param file File to check.
786      * @param passes Whether validation is expected to pass.
787      * @return String containing validation results in textual form.
788      * @throws IOException In case of error
789      */
790     @SuppressWarnings("PMD.JUnitAssertionsShouldIncludeMessage")
791     private Collection<Violation> runValidation(final String file,
792         final boolean passes) throws IOException {
793         final Environment.Mock mock = new Environment.Mock();
794         final File license = this.rule.savePackageInfo(
795             new File(mock.basedir(), CheckstyleValidatorTest.DIRECTORY)
796         ).withLines(CheckstyleValidatorTest.LICENSE)
797             .withEol("\n").file();
798         final Environment env = mock.withParam(
799             CheckstyleValidatorTest.LICENSE_PROP,
800             this.toUrl(license)
801         )
802             .withFile(
803                 String.format("src/main/java/foo/%s", file),
804                 new IoCheckedText(
805                     new TextOf(
806                         new ResourceOf(
807                             new FormattedText("com/qulice/checkstyle/%s", file)
808                         )
809                     )
810                 ).asString()
811             );
812         final Collection<Violation> results =
813             new CheckstyleValidator(env).validate(
814                 env.files(file)
815             );
816         if (passes) {
817             MatcherAssert.assertThat(
818                 results,
819                 Matchers.<Violation>empty()
820             );
821         } else {
822             MatcherAssert.assertThat(
823                 results,
824                 Matchers.not(Matchers.<Violation>empty())
825             );
826         }
827         return results;
828     }

830     /**
831      * Validation results matcher.
832      *
833      * @since 0.1
834      */
835     private static final class ViolationMatcher extends
836         TypeSafeMatcher<Violation> {
837 
838         /**
839          * Message to check.
840          */
841         private final String message;
842 
843         /**
844          * File to check.
845          */
846         private final String file;
847 
848         /**
849          * Expected line.
850          */
851         private final String line;
852 
853         /**
854          * Check name.
855          */
856         private final String check;
857 
858         /**
859          * Constructor.
860          * @param message Message to check
861          * @param file File to check
862          * @param line Line to check
863          * @param check Check name
864          * @checkstyle ParameterNumber (3 lines)
865          */
866         ViolationMatcher(final String message, final String file,
867             final String line, final String check) {
868             super();
869             this.message = message;
870             this.file = file;
871             this.line = line;
872             this.check = check;
873         }
874 
875         /**
876          * Constructor.
877          * @param message Message to check
878          * @param file File to check
879          */
880         ViolationMatcher(final String message, final String file) {
881             this(message, file, "", "");
882         }
883 
884         @Override
885         public boolean matchesSafely(final Violation item) {
886             return item.message().contains(this.message)
887                 && item.file().endsWith(this.file)
888                 && this.lineMatches(item)
889                 && this.checkMatches(item);
890         }
891 
892         @Override
893         public void describeTo(final Description description) {
894             description.appendText("doesn't match");
895         }
896 
897         /**
898          * Check name matches.
899          * @param item Item to check.
900          * @return True if check name matches.
901          */
902         private boolean checkMatches(final Violation item) {
903             return this.check.isEmpty()
904                 || !this.check.isEmpty() && item.name().equals(this.check);
905         }
906 
907         /**
908          * Check that given line matches.
909          * @param item Item to check.
910          * @return True if line matches.
911          */
912         private boolean lineMatches(final Violation item) {
913             return this.line.isEmpty()
914                 || !this.line.isEmpty() && item.lines().equals(this.line);
915         }
916 
917     }
