97       * CheckstyleValidator can catch checkstyle violations.
98       * @throws Exception If something wrong happens inside
99       */
100     @Test
101     public void catchesCheckstyleViolationsInLicense() throws Exception {
102         final Environment.Mock mock = new Environment.Mock();
103         final File license = this.rule.savePackageInfo(
104             new File(mock.basedir(), CheckstyleValidatorTest.DIRECTORY)
105         ).withLines("License-1.", "", "License-2.")
106             .withEol("\n")
107             .file();
108         final String content =
109             // @checkstyle StringLiteralsConcatenation (4 lines)
110             // @checkstyle RegexpSingleline (1 line)
111             "/" + "**\n * License-3.\n *\n * License-2.\n */\n"
112                 + "package foo;\n"
113                 + "public class Foo { }\n";
114         final String name = "Foo.java";
115         final Environment env = mock.withParam(
116             CheckstyleValidatorTest.LICENSE_PROP,
117             this.toUrl(license)
118         ).withFile(String.format("src/main/java/foo/%s", name), content);
119         final Collection<Violation> results =
120             new CheckstyleValidator(env)
121                 .validate(env.files(name));
122         MatcherAssert.assertThat(
123             results,
124             Matchers.hasItem(
125                 new ViolationMatcher(
126                     "Line does not match expected header line of", name
127                 )
128             )
129         );
130     }

781         return String.format("file:%s", file);
782     }

784     /**
785      * Validates that checkstyle reported given violation.
786      * @param file File to check.
787      * @param result Expected validation result.
788      * @param message Message to match
789      * @throws Exception In case of error
790      */
791     private void validate(final String file, final boolean result,
792         final String message) throws Exception {
793         MatcherAssert.assertThat(
794             this.runValidation(file, result),
795             Matchers.hasItem(
796                 new ViolationMatcher(
797                     message, file
798                 )
799             )
800         );
801     }

803     /**
804      * Returns string with Checkstyle validation results.
805      * @param file File to check.
806      * @param passes Whether validation is expected to pass.
807      * @return String containing validation results in textual form.
808      * @throws IOException In case of error
809      */
810     private Collection<Violation> runValidation(final String file,
811         final boolean passes) throws IOException {
812         final Environment.Mock mock = new Environment.Mock();
813         final File license = this.rule.savePackageInfo(
814             new File(mock.basedir(), CheckstyleValidatorTest.DIRECTORY)
815         ).withLines(CheckstyleValidatorTest.LICENSE)
816             .withEol("\n").file();
817         final Environment env = mock.withParam(
818             CheckstyleValidatorTest.LICENSE_PROP,
819             this.toUrl(license)
820         )
821             .withFile(
822                 String.format("src/main/java/foo/%s", file),
823                 new IoCheckedText(
824                     new TextOf(
825                         new ResourceOf(
826                             new FormattedText("com/qulice/checkstyle/%s", file)
827                         )
828                     )
829                 ).asString()
830             );
831         final Collection<Violation> results =
832             new CheckstyleValidator(env).validate(
833                 env.files(file)
834             );
835         if (passes) {
836             MatcherAssert.assertThat(
837                 results,
838                 Matchers.<Violation>empty()
839             );
840         } else {
841             MatcherAssert.assertThat(
842                 results,
843                 Matchers.not(Matchers.<Violation>empty())
844             );
845         }
846         return results;
847     }

849     /**
850      * Validation results matcher.
851      *
852      * @since 0.1
853      */
854     private static final class ViolationMatcher extends
855         TypeSafeMatcher<Violation> {
856 
857         /**
858          * Message to check.
859          */
860         private final String message;
861 
862         /**
863          * File to check.
864          */
865         private final String file;
866 
867         /**
868          * Expected line.
869          */
870         private final String line;
871 
872         /**
873          * Check name.
874          */
875         private final String check;
876 
877         /**
878          * Constructor.
879          * @param message Message to check
880          * @param file File to check
881          * @param line Line to check
882          * @param check Check name
883          * @checkstyle ParameterNumber (3 lines)
884          */
885         ViolationMatcher(final String message, final String file,
886             final String line, final String check) {
887             super();
888             this.message = message;
889             this.file = file;
890             this.line = line;
891             this.check = check;
892         }
893 
894         /**
895          * Constructor.
896          * @param message Message to check
897          * @param file File to check
898          */
899         ViolationMatcher(final String message, final String file) {
900             this(message, file, "", "");
901         }
902 
903         @Override
904         public boolean matchesSafely(final Violation item) {
905             return item.message().contains(this.message)
906                 && item.file().endsWith(this.file)
907                 && this.lineMatches(item)
908                 && this.checkMatches(item);
909         }
910 
911         @Override
912         public void describeTo(final Description description) {
913             description.appendText("doesn't match");
914         }
915 
916         /**
917          * Check name matches.
918          * @param item Item to check.
919          * @return True if check name matches.
920          */
921         private boolean checkMatches(final Violation item) {
922             return this.check.isEmpty()
923                 || !this.check.isEmpty() && item.name().equals(this.check);
924         }
925 
926         /**
927          * Check that given line matches.
928          * @param item Item to check.
929          * @return True if line matches.
930          */
931         private boolean lineMatches(final Violation item) {
932             return this.line.isEmpty()
933                 || !this.line.isEmpty() && item.lines().equals(this.line);
934         }
935 
936     }
