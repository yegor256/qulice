21  import org.cactoos.text.TextOf;

24  
25  /**
26   * Test case for {@link DefaultMavenEnvironment} class methods that
27   * exclude files from validation.
28   * @since 0.19
29   */
30  final class ValidationExclusionTest {
31      /**
32       * Temporary directory for the project source folder.
33       */
34      private static final String TEMP_DIR = "src";
35  
36      /**
37       * Temporary directory for the project subfolder to be excluded.
38       */
39      private static final String TEMP_SUB = "excl";
40  
41      /**
42       * Java files extension.
43       */
44      private static final String JAVA_EXT = ".java";
45  
46      /**
47       * DefaultMavenEnvironment can exclude a path from PMD validation.
48       * @throws Exception If something wrong happens inside
49       */
50      @Test
51      void excludePathFromPmdValidation() throws Exception {
52          final DefaultMavenEnvironment env = new DefaultMavenEnvironment();
53          final Path dir = Files.createTempDirectory(ValidationExclusionTest.TEMP_DIR);
54          final Path subdir = Files.createTempDirectory(dir, ValidationExclusionTest.TEMP_SUB);
55          final File file = File.createTempFile(
56              "PmdExample", ValidationExclusionTest.JAVA_EXT,
57              subdir.toFile()
58          );
59          final MavenProject project = new MavenProject();
60          project.setFile(subdir.toFile());
61          env.setProject(project);
62          Assertions.assertNotNull(project.getBasedir());
63          final String source = new TextOf(
64              new ResourceOf("com/qulice/maven/ValidationExclusion/PmdExample.txt")
65          ).asString();
66          FileUtils.forceDeleteOnExit(file);
67          FileUtils.writeStringToFile(
68              file,
69              source,
70              StandardCharsets.UTF_8
71          );
72          env.setExcludes(
73              Collections.singletonList(
74                  String.format("pmd:/%s/.*", subdir.getFileName())
75              )
76          );
77          final PmdValidator validator = new PmdValidator(env);
78          final Collection<DataSource> files = validator.getNonExcludedFiles(
79              Collections.singletonList(file)
80          );
81          Assertions.assertTrue(files.isEmpty());
82      }
83  
84      /**
85       * DefaultMavenEnvironment can exclude a path from Checkstyle validation.
86       * @throws Exception If something wrong happens inside
87       */
88      @Test
89      void excludePathFromCheckstyleValidation() throws Exception {
90          final DefaultMavenEnvironment env = new DefaultMavenEnvironment();
91          final Path dir = Files.createTempDirectory(ValidationExclusionTest.TEMP_DIR);
92          final Path subdir = Files.createTempDirectory(dir, ValidationExclusionTest.TEMP_SUB);
93          final File file = File.createTempFile(
94              "CheckstyleExample", ValidationExclusionTest.JAVA_EXT,
95              subdir.toFile()
96          );
97          final Build build = new Build();
98          build.setOutputDirectory(dir.toString());
99          final MavenProject project = new MavenProject();
100         project.setFile(subdir.toFile());
101         project.setBuild(build);
102         env.setProject(project);
103         Assertions.assertNotNull(project.getBasedir());
104         Assertions.assertNotNull(env.tempdir());
105         final String source = new TextOf(
106             new ResourceOf("com/qulice/maven/ValidationExclusion/CheckstyleExample.txt")
107         ).asString();
108         FileUtils.forceDeleteOnExit(file);
109         FileUtils.writeStringToFile(
110             file,
111             source,
112             StandardCharsets.UTF_8
113         );
114         env.setExcludes(
115             Collections.singletonList(
116                 String.format("checkstyle:/%s/.*", subdir.getFileName())
117             )
118         );
119         final CheckstyleValidator validator = new CheckstyleValidator(env);
120         final List<File> files = validator.getNonExcludedFiles(Collections.singletonList(file));
121         Assertions.assertTrue(files.isEmpty());
122     }
123 }
