32  import com.google.common.base.Joiner;
33  import com.qulice.spi.Environment;
34  import com.qulice.spi.ValidationException;
35  import java.util.HashSet;
36  import java.util.Set;
37  import org.apache.maven.artifact.Artifact;
38  import org.apache.maven.plugin.testing.stubs.ArtifactStub;
39  import org.apache.maven.project.MavenProject;
40  import org.apache.maven.shared.dependency.analyzer.ProjectDependencyAnalysis;
41  import org.apache.maven.shared.dependency.analyzer.ProjectDependencyAnalyzer;
42  import org.junit.jupiter.api.Assertions;
43  import org.junit.jupiter.api.Test;
44  import org.mockito.Mockito;
45  
46  /**
47   * Test case for {@link DependenciesValidator} class.
48   * @since 0.3
49   */
50  public final class DependenciesValidatorTest {
51  
52      /**
53       * Plexus role.
54       */
55      private static final String ROLE = ProjectDependencyAnalyzer.ROLE;
56  
57      /**
58       * Plexus hint.
59       */
60      private static final String HINT = "default";
61  
62      /**
63       * Compile scope.
64       */
65      private static final String SCOPE = "compile";
66  
67      /**
68       * Jar type.
69       */
70      private static final String TYPE = "jar";
71  
72      /**
73       * DependencyValidator can pass on when no violations are found.
74       * @throws Exception If something wrong happens inside
75       */
76      @Test
77      public void passesIfNoDependencyProblemsFound() throws Exception {
78          final ProjectDependencyAnalysis analysis =
79              Mockito.mock(ProjectDependencyAnalysis.class);
80          final ProjectDependencyAnalyzer analyzer = this.analyzer(analysis);
81          final MavenEnvironment env = new MavenEnvironmentMocker().inPlexus(
82              DependenciesValidatorTest.ROLE,
83              DependenciesValidatorTest.HINT,
84              analyzer
85          ).mock();
86          new DependenciesValidator().validate(env);
87      }
88  
89      /**
90       * DependencyValidator can catch dependency problems.
91       * @throws Exception If something wrong happens inside
92       */
93      @Test
94      public void catchesDependencyProblemsAndThrowsException() throws Exception {
95          final ProjectDependencyAnalysis analysis =
96              Mockito.mock(ProjectDependencyAnalysis.class);
97          final Set<Artifact> unused = new HashSet<>();
98          unused.add(Mockito.mock(Artifact.class));
99          Mockito.doReturn(unused).when(analysis).getUsedUndeclaredArtifacts();
100         final ProjectDependencyAnalyzer analyzer = this.analyzer(analysis);
101         final MavenEnvironment env = new MavenEnvironmentMocker().inPlexus(
102             DependenciesValidatorTest.ROLE,
103             DependenciesValidatorTest.HINT,
104             analyzer
105         ).mock();
106         Assertions.assertThrows(
107             ValidationException.class,
108             () -> new DependenciesValidator().validate(env)
109         );
110     }
111 
112     /**
113      * DependencyValidator can ignore runtime scope dependencies.
114      * @throws Exception If something wrong happens inside
115      */
116     @Test
117     public void ignoresRuntimeScope() throws Exception {
118         final ProjectDependencyAnalysis analysis =
119             Mockito.mock(ProjectDependencyAnalysis.class);
120         final Artifact artifact = Mockito.mock(Artifact.class);
121         final Set<Artifact> unused = new HashSet<>();
122         unused.add(artifact);
123         Mockito.doReturn(unused).when(analysis).getUnusedDeclaredArtifacts();
124         Mockito.doReturn(Artifact.SCOPE_RUNTIME).when(artifact).getScope();
125         final ProjectDependencyAnalyzer analyzer = this.analyzer(analysis);
126         final MavenEnvironment env = new MavenEnvironmentMocker().inPlexus(
127             DependenciesValidatorTest.ROLE,
128             DependenciesValidatorTest.HINT,
129             analyzer
130         ).mock();
131         new DependenciesValidator().validate(env);
132     }
133 
134     /**
135      * DependencyValidator can exclude used undeclared dependencies.
136      * @throws Exception If something wrong happens inside
137      */
138     @Test
139     public void excludesUsedUndeclaredDependencies() throws Exception {
140         final ProjectDependencyAnalysis analysis =
141             Mockito.mock(ProjectDependencyAnalysis.class);
142         final Set<Artifact> used = new HashSet<>();
143         final ArtifactStub artifact = new ArtifactStub();
144         artifact.setGroupId("group");
145         artifact.setArtifactId("artifact");
146         artifact.setScope(DependenciesValidatorTest.SCOPE);
147         artifact.setVersion("2.3.4");
148         artifact.setType(DependenciesValidatorTest.TYPE);
149         used.add(artifact);
150         Mockito.doReturn(used).when(analysis).getUsedUndeclaredArtifacts();
151         final ProjectDependencyAnalyzer analyzer = this.analyzer(analysis);
152         final MavenEnvironment env = new MavenEnvironmentMocker().inPlexus(
153             DependenciesValidatorTest.ROLE,
154             DependenciesValidatorTest.HINT,
155             analyzer
156         ).mock();
157         new DependenciesValidator().validate(
158             new MavenEnvironment.Wrap(
159                 new Environment.Mock().withExcludes(
160                     Joiner.on(':').join(
161                         artifact.getGroupId(), artifact.getArtifactId()
162                     )
163                 ), env
164             )
165         );
166     }
167 
168     /**
169      * DependencyValidator can exclude unused declared dependencies.
170      * @throws Exception If something wrong happens inside
171      */
172     @Test
173     public void excludesUnusedDeclaredDependencies() throws Exception {
174         final ProjectDependencyAnalysis analysis =
175             Mockito.mock(ProjectDependencyAnalysis.class);
176         final Set<Artifact> unused = new HashSet<>();
177         final ArtifactStub artifact = new ArtifactStub();
178         artifact.setGroupId("othergroup");
179         artifact.setArtifactId("otherartifact");
180         artifact.setScope(DependenciesValidatorTest.SCOPE);
181         artifact.setVersion("1.2.3");
182         artifact.setType(DependenciesValidatorTest.TYPE);
183         unused.add(artifact);
184         Mockito.doReturn(unused).when(analysis).getUnusedDeclaredArtifacts();
185         final ProjectDependencyAnalyzer analyzer = this.analyzer(analysis);
186         final MavenEnvironment env = new MavenEnvironmentMocker().inPlexus(
187             DependenciesValidatorTest.ROLE,
188             DependenciesValidatorTest.HINT,
189             analyzer
190         ).mock();
191         new DependenciesValidator().validate(
192             new MavenEnvironment.Wrap(
193                 new Environment.Mock().withExcludes(
194                     Joiner.on(':').join(
195                         artifact.getGroupId(), artifact.getArtifactId()
196                     )
197                 ), env
198             )
199         );
200     }
201 
202     /**
203      * Create analyzer object.
204      * @param analysis The analysis object
205      * @return The object
206      * @throws Exception If something wrong happens inside
207      */
208     private ProjectDependencyAnalyzer analyzer(
209         final ProjectDependencyAnalysis analysis) throws Exception {
210         final ProjectDependencyAnalyzer analyzer =
211             Mockito.mock(ProjectDependencyAnalyzer.class);
212         Mockito.doReturn(analysis).when(analyzer)
213             .analyze(Mockito.any(MavenProject.class));
214         return analyzer;
215     }
