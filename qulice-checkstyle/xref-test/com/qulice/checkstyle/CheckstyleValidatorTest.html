781     /**
782      * Returns string with Checkstyle validation results.
783      * @param file File to check.
784      * @param passes Whether validation is expected to pass.
785      * @return String containing validation results in textual form.
786      * @throws IOException In case of error
787      */
788     @SuppressWarnings("PMD.JUnitAssertionsShouldIncludeMessage")
789     private Collection<Violation> runValidation(final String file,
790         final boolean passes) throws IOException {
791         final Environment.Mock mock = new Environment.Mock();
792         final File license = this.rule.savePackageInfo(
793             new File(mock.basedir(), CheckstyleValidatorTest.DIRECTORY)

828     /**
829      * Validation results matcher.
830      *
831      * @since 0.1
832      */
833     private static final class ViolationMatcher extends
834         TypeSafeMatcher<Violation> {
835 
836         /**
837          * Message to check.
838          */
839         private final String message;
840 
841         /**
842          * File to check.
843          */
844         private final String file;
845 
846         /**
847          * Expected line.
848          */
849         private final String line;
850 
851         /**
852          * Check name.
853          */
854         private final String check;
855 
856         /**
857          * Constructor.
858          * @param message Message to check
859          * @param file File to check
860          * @param line Line to check
861          * @param check Check name
862          * @checkstyle ParameterNumber (3 lines)
863          */
864         ViolationMatcher(final String message, final String file,
865             final String line, final String check) {
866             super();
867             this.message = message;
868             this.file = file;
869             this.line = line;
870             this.check = check;
871         }
872 
873         /**
874          * Constructor.
875          * @param message Message to check
876          * @param file File to check
877          */
878         ViolationMatcher(final String message, final String file) {
879             this(message, file, "", "");
880         }
881 
882         @Override
883         public boolean matchesSafely(final Violation item) {
884             return item.message().contains(this.message)
885                 && item.file().endsWith(this.file)
886                 && this.lineMatches(item)
887                 && this.checkMatches(item);
888         }
889 
890         @Override
891         public void describeTo(final Description description) {
892             description.appendText("doesn't match");
893         }
894 
895         /**
896          * Check name matches.
897          * @param item Item to check.
898          * @return True if check name matches.
899          */
900         private boolean checkMatches(final Violation item) {
901             return this.check.isEmpty()
902                 || !this.check.isEmpty() && item.name().equals(this.check);
903         }
904 
905         /**
906          * Check that given line matches.
907          * @param item Item to check.
908          * @return True if line matches.
909          */
910         private boolean lineMatches(final Violation item) {
911             return this.line.isEmpty()
912                 || !this.line.isEmpty() && item.lines().equals(this.line);
913         }
914 
915     }
